## 三、序列特征增强

### 3.1 序列统计特征

```python
# 对每个domain的序列特征进行统计
domains = ['a', 'b', 'c', 'd']

for domain in domains:
    seq_cols = [col for col in df.columns if f'domain_{domain}_seq_' in col]
    
    # 序列长度（非零元素个数）
    df[f'domain_{domain}_seq_len'] = (df[seq_cols] != 0).sum(axis=1)
    
    # 序列最大值、最小值、平均值
    df[f'domain_{domain}_seq_max'] = df[seq_cols].max(axis=1)
    df[f'domain_{domain}_seq_min'] = df[seq_cols].min(axis=1)
    df[f'domain_{domain}_seq_mean'] = df[seq_cols].mean(axis=1)
    df[f'domain_{domain}_seq_std'] = df[seq_cols].std(axis=1)
    
    # 序列中最后一个非零元素（最近行为）
    df[f'domain_{domain}_seq_last'] = df[seq_cols].apply(
        lambda x: x[x != 0].iloc[-1] if (x != 0).any() else 0, axis=1
    )
```


### 3.2 序列与目标物品的匹配特征

```python
# 目标物品是否在历史序列中出现
for domain in ['a', 'b', 'c', 'd']:
    seq_cols = [col for col in df.columns if f'domain_{domain}_seq_' in col]
    df[f'item_in_domain_{domain}'] = df.apply(
        lambda row: int(row['item_id'] in row[seq_cols].values), axis=1
    )
    
    # 目标物品在序列中的位置（越靠后越近期）
    df[f'item_pos_in_domain_{domain}'] = df.apply(
        lambda row: np.where(row[seq_cols].values == row['item_id'])[0][-1] 
        if row['item_id'] in row[seq_cols].values else -1, axis=1
    )
```

### 3.3 跨域序列特征

```python
# 用户在不同域的活跃度对比
df['total_seq_len'] = sum(df[f'domain_{d}_seq_len'] for d in ['a', 'b', 'c', 'd'])
df['domain_a_ratio'] = df['domain_a_seq_len'] / (df['total_seq_len'] + 1)
df['domain_b_ratio'] = df['domain_b_seq_len'] / (df['total_seq_len'] + 1)
df['domain_c_ratio'] = df['domain_c_seq_len'] / (df['total_seq_len'] + 1)
df['domain_d_ratio'] = df['domain_d_seq_len'] / (df['total_seq_len'] + 1)

# 用户域偏好熵（衡量用户是否专注于某个域）
from scipy.stats import entropy
df['domain_entropy'] = df[['domain_a_ratio', 'domain_b_ratio', 
                            'domain_c_ratio', 'domain_d_ratio']].apply(
    lambda x: entropy(x + 1e-10), axis=1
)
```

## 八、目标编码（Target Encoding）

### 8.1 基础目标编码

```python
from sklearn.model_selection import KFold

def target_encode(df, col, target='label_type', n_splits=5):
    """K-fold目标编码，避免过拟合"""
    kf = KFold(n_splits=n_splits, shuffle=True, random_state=42)
    df[f'{col}_target_enc'] = 0
    
    for train_idx, val_idx in kf.split(df):
        # 计算训练集的目标均值
        target_mean = df.iloc[train_idx].groupby(col)[target].mean()
        # 映射到验证集
        df.loc[val_idx, f'{col}_target_enc'] = df.loc[val_idx, col].map(target_mean)
    
    # 填充缺失值（全局均值）
    global_mean = df[target].mean()
    df[f'{col}_target_enc'].fillna(global_mean, inplace=True)
    
    return df

# 对关键类别特征进行目标编码
for col in ['user_id', 'item_id', 'user_int_feats_1', 'item_int_feats_5']:
    df = target_encode(df, col)
```


### 8.2 平滑目标编码

```python
def smooth_target_encode(df, col, target='label_type', alpha=10):
    """带平滑的目标编码，对低频类别更稳健"""
    global_mean = df[target].mean()
    agg = df.groupby(col)[target].agg(['sum', 'count'])
    
    # 贝叶斯平滑
    smooth_enc = (agg['sum'] + alpha * global_mean) / (agg['count'] + alpha)
    df[f'{col}_smooth_enc'] = df[col].map(smooth_enc)
    df[f'{col}_smooth_enc'].fillna(global_mean, inplace=True)
    
    return df
```

## 十一、特征工程代码模板

### 完整特征构建流程

```python
import pandas as pd
import numpy as np
from sklearn.model_selection import KFold

def build_features(df):
    """完整的特征工程流程"""
    
    # 1. 时间特征
    df['datetime'] = pd.to_datetime(df['timestamp'], unit='s')
    df['hour'] = df['datetime'].dt.hour
    df['day_of_week'] = df['datetime'].dt.dayofweek
    df['is_weekend'] = df['day_of_week'].isin([5, 6]).astype(int)
    
    # 2. 统计特征
    # 用户CTR
    user_ctr = df.groupby('user_id')['label_type'].agg(['sum', 'count', 'mean']).reset_index()
    user_ctr.columns = ['user_id', 'user_click_sum', 'user_expo_count', 'user_ctr']
    df = df.merge(user_ctr, on='user_id', how='left')
    
    # 物品CTR
    item_ctr = df.groupby('item_id')['label_type'].agg(['sum', 'count', 'mean']).reset_index()
    item_ctr.columns = ['item_id', 'item_click_sum', 'item_expo_count', 'item_ctr']
    df = df.merge(item_ctr, on='item_id', how='left')
    
    # 3. 交叉特征
    df['user_item'] = df['user_id'].astype(str) + '_' + df['item_id'].astype(str)
    df['u1_i5'] = df['user_int_feats_1'].astype(str) + '_' + df['item_int_feats_5'].astype(str)
    
    # 4. 序列特征
    for domain in ['a', 'b', 'c', 'd']:
        seq_cols = [col for col in df.columns if f'domain_{domain}_seq_' in col]
        df[f'domain_{domain}_seq_len'] = (df[seq_cols] != 0).sum(axis=1)
        df[f'domain_{domain}_seq_mean'] = df[seq_cols].mean(axis=1)
        df[f'domain_{domain}_seq_max'] = df[seq_cols].max(axis=1)
    
    # 5. 稠密特征统计
    dense_cols = [col for col in df.columns if 'user_dense_feats_' in col]
    df['user_dense_sum'] = df[dense_cols].sum(axis=1)
    df['user_dense_mean'] = df[dense_cols].mean(axis=1)
    
    return df

# 使用
df = build_features(df)
```

## 十二、注意事项

### 12.1 时间泄露问题

⚠️ **重要**：构建统计特征时，必须避免使用未来信息！

```python
# ❌ 错误：使用全局统计（包含未来数据）
user_ctr = df.groupby('user_id')['label_type'].mean()

# ✅ 正确：按时间排序，使用历史数据
df = df.sort_values('timestamp')
df['user_ctr'] = df.groupby('user_id')['label_type'].expanding().mean().shift(1)
```

### 12.2 特征数量控制

- 不要一次性加入所有特征，容易过拟合
- 建议分批实验，每次加入5-10个特征
- 使用特征重要性分析，删除低价值特征

### 12.3 类别特征基数控制

```python
# 对高基数类别特征进行频率过滤
def filter_rare_categories(df, col, min_count=10):
    value_counts = df[col].value_counts()
    rare_values = value_counts[value_counts < min_count].index
    df[col] = df[col].replace(rare_values, -1)  # 用-1表示稀有类别
    return df
```

## 十三、实验建议

### 实验流程

1. **Baseline**：只用原始特征跑一个baseline
2. **+统计特征**：加入用户CTR、物品CTR
3. **+交叉特征**：加入user×item交叉
4. **+序列特征**：加入序列统计
5. **+时间特征**：加入时间相关特征
6. **+目标编码**：加入目标编码特征

每一步记录AUC提升，找到最有效的特征类型。
